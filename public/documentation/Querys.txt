Query para crear tu tabla:

CREATE TABLE planets (
    id VARCHAR PRIMARY KEY,
    name VARCHAR NOT NULL,
    englishName VARCHAR NOT NULL,
    image VARCHAR NOT NULL,
    isPlanet BOOLEAN NOT NULL,
    moons SMALLINT, -- Número de lunas
    semimajorAxis BIGINT, -- Distancia semimayor en kilómetros
    perihelion BIGINT, -- Perihelio en kilómetros
    aphelion BIGINT, -- Afelio en kilómetros
    eccentricity FLOAT, -- Excentricidad orbital
    inclination FLOAT, -- Inclinación orbital en grados
    massValue FLOAT, -- Valor de la masa
    massExponent SMALLINT, -- Exponente de la masa (científica)
    volValue FLOAT, -- Valor del volumen
    volExponent SMALLINT, -- Exponente del volumen
    density FLOAT, -- Densidad en g/cm³
    gravity FLOAT, -- Gravedad en m/s²
    escape FLOAT, -- Velocidad de escape en m/s
    meanRadius FLOAT, -- Radio medio en km
    equaRadius FLOAT, -- Radio ecuatorial en km
    polarRadius FLOAT, -- Radio polar en km
    flattening FLOAT, -- Aplanamiento
    dimension VARCHAR, -- Dimensiones del planeta
    sideralOrbit FLOAT, -- Órbita sideral en días
    sideralRotation FLOAT, -- Rotación sideral en horas
    aroundPlanet VARCHAR, -- Planeta alrededor del cual orbita (si es aplicable)
    discoveredBy VARCHAR, -- Descubridor
    discoveryDate DATE, -- Fecha de descubrimiento
    alternativeName VARCHAR, -- Nombres alternativos
    axialTilt FLOAT, -- Inclinación axial en grados
    avgTemp FLOAT, -- Temperatura promedio en Celsius
    mainAnomaly FLOAT, -- Anomalía media en grados
    argPeriapsis FLOAT, -- Argumento del periapsis en grados
    longAscNode FLOAT, -- Longitud del nodo ascendente en grados
    bodyType VARCHAR NOT NULL, -- Tipo de cuerpo (e.g., 'Planet')
    rel VARCHAR NOT NULL -- URL relacionada con más información
);
